# Test Case 1: Empty List
>>> my_list = MyList()
>>> print(my_list)
[]
>>> my_list.print_sorted()
[]
>>> print(my_list)
[]

# Test Case 2: List with one element
>>> my_list.append(42)
>>> print(my_list)
[42]
>>> my_list.print_sorted()
[42]
>>> print(my_list)
[42]

# Test Case 3: List with multiple elements, unsorted
>>> my_list = MyList()
>>> my_list.append(5)
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> print(my_list)
[5, 1, 4, 2, 3]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[5, 1, 4, 2, 3]

# Test Case 4: List with already sorted elements
>>> my_list = MyList([1, 2, 3, 4, 5])
>>> print(my_list)
[1, 2, 3, 4, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 2, 3, 4, 5]

# Test Case 5: List with repeated elements
>>> my_list = MyList([3, 1, 4, 1, 5, 9])
>>> print(my_list)
[3, 1, 4, 1, 5, 9]
>>> my_list.print_sorted()
[1, 1, 3, 4, 5, 9]
>>> print(my_list)
[3, 1, 4, 1, 5, 9]
